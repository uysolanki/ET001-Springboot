package com.itv.FirstSpringbootApp.model;

import java.util.List;

import javax.persistence.Column;
import javax.persistence.ElementCollection;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;

import lombok.AllArgsConstructor;
import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;

@Setter
@Getter
@NoArgsConstructor
@AllArgsConstructor
@Entity
public class PetAdoption {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    private String fullName;
    private String email;
    private String phone;

    @Column(length = 500)
    private String address;

    private String petType;

    @ElementCollection
    private List<String> reasons;

    private String breed;

    @Column(length = 1000)
    private String additionalDetails;

    private String proofOfIdentityFilePath;

}

package com.itv.FirstSpringbootApp.controller;

import java.io.File;

import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;

import org.springframework.web.multipart.MultipartFile;

import com.itv.FirstSpringbootApp.model.PetAdoption;
import com.itv.FirstSpringbootApp.services.PetAdoptionService;
@Controller
public class PetAdoptionController {

    @Autowired
    private PetAdoptionService petAdoptionService;


    @RequestMapping("/register-pet")
	public String register(Model model) {
    	PetAdoption petAdoption=new PetAdoption();
		model.addAttribute("myPetAdoption",petAdoption);
		return "adopt-form";
	}
    
   
    @PostMapping("/addAdoption")
    public String addAdoption(@ModelAttribute PetAdoption petAdoption,
			@RequestParam("fileUpload") MultipartFile file) throws IOException {
  
        String uploadDir = System.getProperty("user.dir") + "/uploads"; 

        File directory = new File(uploadDir);
        if (!directory.exists()) {
            directory.mkdirs();
        }

        String filePath = uploadDir + "/" + file.getOriginalFilename();
        file.transferTo(new File(filePath));

        petAdoption.setProofOfIdentityFilePath(filePath);

        PetAdoption savedAdoption = petAdoptionService.savePetAdoption(petAdoption);
        return "submit-adoption";
    }


    @RequestMapping("/adoptions")
    public String getAllAdoptions(Model model) {
        List<PetAdoption> adoption = petAdoptionService.getAllAdoptions();
        model.addAttribute("adoptions", adoption);
        return "adoptions"; 
    }



   @PostMapping("/update")
   public String updateAdoption(@ModelAttribute("adoption") PetAdoption petAdoption, Model model) {
	   petAdoptionService.updateAdoption(petAdoption);
       return "redirect:/home"; 
   }
   
   
}





Service
package com.itv.FirstSpringbootApp.services;

import java.util.List;
import java.util.Optional;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import com.itv.FirstSpringbootApp.model.PetAdoption;
import com.itv.FirstSpringbootApp.repository.PetAdoptionRepository;

@Service
public class PetAdoptionService {

    @Autowired
    private PetAdoptionRepository petAdoptionRepository;

    public PetAdoption savePetAdoption(PetAdoption petAdoption) {
        return petAdoptionRepository.save(petAdoption);
    }

    public List<PetAdoption> getAllAdoptions() {
        return petAdoptionRepository.findAll();
    }

    public Optional<PetAdoption> getAdoptionById(Long id) {
        return petAdoptionRepository.findById(id);
    }

    public void deleteAdoption(Long id) {
        petAdoptionRepository.deleteById(id);
    }

	public void updateAdoption(PetAdoption petAdoption) {
		 if (petAdoption.getId() == null || !petAdoptionRepository.existsById(petAdoption.getId())) {
	            throw new IllegalArgumentException("Adoption record not found!");
	        }
		 petAdoptionRepository.save(petAdoption); 
		
	}
}
